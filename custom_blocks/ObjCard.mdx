export const ObjCard = ({ title, type, description, data, children }) => {
  const baseClasses = `
    bg-[var(--yuno-card-background)]
    !my-[1.5em]
    rounded-[5px]
    border-l-[8px]
    border-l-[color:var(--yuno-purple)]
    !pl-[12px] !pr-[12px] !py-[12px]
  `;

  if (children) {
    return (
      <details className={baseClasses}>
        <summary className="cursor-pointer">
          <strong>
            <code className="!pl-0">{title}</code>
          </strong>
          <small className="ml-1">{type}</small>
          <div className="mt-2">{description}</div>
        </summary>

        <div className="mt-4">{children}</div>
      </details>
    );
  }

  return (
    <div className={baseClasses}>
      <div>
        <strong>
          <code className="!pl-0">{title}</code>
        </strong>
        <small className="ml-1">{type}</small>
      </div>
      <div className="mt-2">{description}</div>
      {data && (
        <div className="mt-2">
          <small>{data}</small>
        </div>
      )}
    </div>
  );
};

<ObjCard title="parameter" description="object description" type="String" data="Examples">
	You can add an internal object here.

	If you add an internal object, the `data` field is not displayed.
</ObjCard>
